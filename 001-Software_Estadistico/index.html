## Iteración de punto fijo Aproximación inicial inicial: x0 = 0.8,
##   Iteración xn_ = g(x_{n-1})
library(animation)
ani.options(interval = 0.5, nmax = 14)
g = function(x) (x + 1) * sin(x^2)
pxa = rep(NA, times = ani.options("nmax") + 1)
pxb = rep(NA, times = ani.options("nmax") + 1)
x0 = 0.8
a = -3
b = 2
xa = x0
xb = g(x0)
pxa[1] = x0
pxb[1] = 0
pxa[2] = xa
pxb[2] = xb
k = 4
while (k <= ani.options("nmax")) {
    pxa[k - 1] = xa
    pxa[k:(k + 2)] = xb
    pxb[(k - 1):(k + 1)] = xb
    pxb[k + 2] = g(xb)
    xa = xb
    xb = g(xa)
    k = k + 2
}
for (j in 1:ani.options("nmax")) {
    plot(g, a, b, col = "blue")
    curve((x), a, b, add = T)
    abline(h = 0, v = 0, lty = 2)
    arrows(pxa[1:j], pxb[1:j], pxa[2:(j + 1)], pxb[2:(j + 
        1)], col = "red", length = 0.15, angle = 20)
    ani.pause()
}
